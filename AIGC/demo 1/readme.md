- js 获取一个值，当值为空时，用一个默认值
    var value = ""; // 假设要获取的值为空

    var defaultValue = "默认值"; // 设置默认值

    var result = value !== "" ? value : defaultValue;

    console.log(result); // 输出：默认值

- 优化上面的代码
    var value = ""; // 假设要获取的值为空
    
    var defaultValue = "默认值"; // 设置默认值

    var result = value || defaultValue;

    console.log(result); // 输出：默认值
